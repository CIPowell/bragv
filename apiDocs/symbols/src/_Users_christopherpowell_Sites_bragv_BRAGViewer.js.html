141 BRAGV.Viewer.prototype = {
142 		draw : function()
143 		{
144 			var thisDraw = new Date().getTime();
145 			this.lastdraw = thisDraw;
146 			this.resetViewer(thisDraw);
147 			this.drawTicks(thisDraw);
148 			this.drawTracks(this.trackWidth, thisDraw);
149 		},
150 		drawTicks : function(thisDraw)
151 		{
152 			if(thisDraw < this.lastDraw) return; 
153 			var start = 40;
154 			var end = this.trackWidth + 20;
155 			
156 			this.ctx.fillStyle = 'rgba(0,0,0,1)';
157 			
158 			this.ctx.beginPath();
159 			this.ctx.moveTo(start, 10);
160 			this.ctx.lineTo(start, 20);
161 			this.ctx.stroke();
162 			
163 			this.ctx.beginPath();
164 			this.ctx.moveTo(end, 10);
165 			this.ctx.lineTo(end, 20);
166 			this.ctx.stroke();
167 			
168 			var txt = (this.offset).toString();
169 			var wt = this.ctx.measureText(txt).width;
170 			this.ctx.fillText(this.offset, start-wt, 10);
171 			var txt = (this.offset + this.numBases).toString();
172 			var wt = this.ctx.measureText(txt).width;
173 			this.ctx.fillText(txt, end-wt, 10);
174 		},
175 		drawTracks : function(w, thisDraw)
176 		{
177 			var i = 1;
178 			var trackheight = 20;
179 			var padding = 3;
180 			var labelWidth = this.labelWidth;
181 			
182 			w = w - trackheight;
183 			
184 			for(var t in this.tracks)
185 			{
186 				if(thisDraw < this.lastDraw) return; 
187 				var tracky  = i*(trackheight + padding);
188 				var isReverse = this.tracks[t].isReverse;
189 				
190 				this.ctx.fillStyle = 'rgba(0, 0, 0, 1)';
191 				this.ctx.fillText(t, 5, ++i * (trackheight + padding) -10);
192 				
193 				this.ctx.fillStyle = 'rgba(80, 80, 80, 0.1)';
194 				this.ctx.fillRect(labelWidth, tracky, w, trackheight);
195 				
196 				
197 				this.lastpos = 0;
198 				var features = this.tracks[t].features;
199 				var count = features.length;
200 				
201 				this.baseWidth = w / this.numBases;
202 				
203 				for(var j = 0; j != count; j++)
204 				{
205 					if(thisDraw < this.lastDraw) return; 
206 					if(features[j].s > this.offset + this.numBases) break;
207 					if(features[j].e < this.offset) continue;
208 					
209 					var start = labelWidth + (Math.max((features[j].s - this.offset) * this.baseWidth, 0));
210 					var width = (Math.min(features[j].e, this.numBases + this.offset) - Math.max(this.offset, features[j].s)) * this.baseWidth;
211 					
212 					if((start+width) < (this.lastpos + 1)) continue;
213 					this.lastpos = start + width;
214 					if(this.selectedFeature[0] == t && this.selectedFeature[1] == j)
215 					{
216 						this.ctx.fillStyle = this.selectedColour;
217 					}
218 					else if(!features[j].c)
219 					{
220 						this.ctx.fillStyle = 'rgba(255, 0, 0, 1)';
221 					}
222 					else
223 					{
224 						this.ctx.fillStyle = features[j].c;
225 					}
226 					this.ctx.fillRect(start, tracky, width, trackheight);
227 					
228 					this.ctx.strokeStyle = 'rgba(0,0,0,0.3)';
229 					this.ctx.strokeRect(start, tracky, width, trackheight);
230 				}
231 				
232 				this.ctx.fillStyle = 'rgba(200,200,200,0.3)';
233 				this.ctx.beginPath();
234 				if(isReverse)
235 				{
236 					this.ctx.moveTo(labelWidth, tracky);
237 					this.ctx.lineTo(labelWidth * 0.7, tracky + (trackheight * .5));
238 					this.ctx.lineTo(labelWidth, tracky + trackheight);
239 					this.ctx.closePath();
240 					this.ctx.fill();
241 					
242 					this.ctx.beginPath();
243 					var x = labelWidth + w;
244 					this.ctx.moveTo(x, tracky);
245 					this.ctx.lineTo(x - labelWidth * 0.3, tracky + (trackheight * .5));
246 					this.ctx.lineTo(x, tracky + trackheight);
247 					this.ctx.closePath();
248 					this.ctx.fill();
249 				}
250 				else
251 				{
252 					this.ctx.moveTo(labelWidth, tracky);
253 					this.ctx.lineTo(labelWidth * 1.3, tracky + (trackheight * .5));
254 					this.ctx.lineTo(labelWidth, tracky + trackheight);
255 					this.ctx.closePath();
256 					this.ctx.fill();
257 					
258 					this.ctx.beginPath();
259 					var x = labelWidth + w;
260 					this.ctx.moveTo(x, tracky);
261 					this.ctx.lineTo(x + labelWidth * 0.3, tracky + (trackheight * .5));
262 					this.ctx.lineTo(x, tracky + trackheight);
263 					this.ctx.closePath();
264 					this.ctx.fill();
265 				}
266 			}
267 		},
268 		loadAnnotation : function(url)
269 		{
270 			var viewer = this;
271 			$.getJSON(url, null, function(data)
272 			{	
273 				 viewer.addTracks(data);
274 			});
275 		},
276 		addTracks : function(obj)
277 		{
278 			var viewer = this;
279 			var count = obj.length;
280 			for(var i = 0; i < count; i++)
281 			{
282 				this.c_length = Math.max(obj[i].length, this.c_length);
283 				var id = obj[i].frame + '_' + obj[i].strand;
284 				
285 				if(!viewer.tracks[id])viewer.tracks[id] = new BRAGV.Track(id);
286 				viewer.tracks[id].isReverse = obj[i].strand < 0;
287 				viewer.tracks[id].features = obj[i].features;
288 				viewer.trackIndex[i] = id;
289 			}
290 			this.numBases = this.c_length;
291 			this.draw();
292 		},
293 		getTrackAt : function(x,y)
294 		{
295 			
296 		},
297 		getBaseAt : function(x,y)
298 		{
299 			
300 		},
301 		getFeatureAt : function(x,y)
302 		{
303 			
304 		},
305 		resetViewer : function()
306 		{
307 			this.ctx.clearRect(0,0, this.trackWidth + this.labelWidth, 400);
308 		},
309 		setOffset : function(val)
310 		{
311 			clearTimeout(this.nextDraw);
312 			var d = this;
313 			
314 			val = Math.round((val / this.horizontalScroller.max) * (this.c_length - this.numBases));
315 			
316 			this.offset = Math.round(val);
317 			this.nextDraw = setTimeout(function(){d.draw();}, this.numBases > 1000000 ? 100 : 10);
318 		},
319 		setZoom : function(val)
320 		{
321 			clearTimeout(this.nextDraw);
322 			var d = this;
323 			this.numBases = Math.pow(10,Math.round(val / 20) + 1);
324 			this.nextDraw = setTimeout(function(){d.draw();}, this.numBases > 1000000 ? 100 : 10);
325 		}
326 };

355 BRAGV.Scroller = function(divName, dir)
356 {
357 	this.div = $('#' + divName);
358 
359 	this.direction = dir;
360 	this.min = 0;
361 	this.max = 100;
362 	this.step = 10;
363 	
364 	if(dir == 'v')
365 	{
366 		this.div.prepend('<div class="bragv_scroller bragv_vertical"><div class="bragv_scroller_handle"></div><div class="bragv_scroller_ctl_lesser"></div><div class="bragv_scroller_ctl_greater"></div></div>');
367 		$('.bragv_vertical').height($('.bragv_canvas').height() - 27);
368 	}
369 	else
370 	{
371 		this.div.append('<div class="bragv_scroller bragv_horizontal"><div class="bragv_scroller_handle"></div><div class="bragv_scroller_ctl_lesser"></div><div class="bragv_scroller_ctl_greater"></div></div>');
372 	}
373 	this.base = $('.bragv_scroller', this.div);
374 	this.base = this.base[0];
375 	
376 	var bar = this;
377 	$('.bragv_scroller_ctl_lesser', this.base).click(function(evt){
378 		var newval = bar.getValue() - bar.step;
379 		newval = newval - (newval % bar.step);
380 		newval = Math.max(newval, bar.min);
381 		bar.setValue(newval);
382 	});
383 	$('.bragv_scroller_ctl_greater', this.base).click(function(evt){
384 		var newval = bar.getValue() + bar.step;
385 		newval = newval + (newval % bar.step);
386 		newval = Math.min(newval, bar.max);
387 		bar.setValue(newval);
388 	});
389 	$('.bragv_scroller_handle', this.base).draggable({
390 		containment : 'parent',
391 		drag : function(event, ui)
392 		{
393 			bar.setValue(bar.getValue());
394 			//console.debug(bar.getValue());
395 		}
396 	});
397 	this.setValue(0);
398 };

400 BRAGV.Scroller.prototype = {
401 	getValue: function()
402 	{
403 		var rawval;
404 		var d_range;
405 		var v_range = this.max - this.min;
406 		
407 		if(this.direction == 'v')
408 		{ 
409 			d_range = $(this.base).innerHeight() - 24; // inner width of slider div minus width of slider
410 			rawval = $('.bragv_scroller_handle', this.base).css('top').replace('px','');
411 			rawval = d_range - rawval;
412 		}
413 		else
414 		{ 
415 			d_range = $(this.base).innerWidth() - 24;
416 			rawval = $('.bragv_scroller_handle', this.base).css('left').replace('px',''); 
417 		}
418 		 
419 		var val = (rawval/d_range) * v_range;
420 		val += this.min;
421 		return val;
422 	},
423 	setValue : function(val)
424 	{
425 		var rawval;
426 		var d_range;
427 		var v_range = this.max - this.min;
428 		var o_val = val;
429 		if(this.direction == 'v')
430 		{ 
431 			val = this.max - val;
432 			d_range = $(this.base).innerHeight() - 24; // inner width of slider div minus width of slider
433 		}
434 		else
435 		{ 
436 			d_range = $(this.base).innerWidth() - 24;
437 		}
438 		
439 		rawval = (val / v_range) * d_range;
440 		if(this.direction == 'v')
441 		{ 
442 			$('.bragv_scroller_handle', this.base).css('top', rawval + 'px'); 
443 		}
444 		else
445 		{ 
446 			$('.bragv_scroller_handle', this.base).css('left', rawval + 'px'); 
447 		}
448 		
449 		if(this.onValueChange) this.onValueChange(o_val);
450 	}
451 };
